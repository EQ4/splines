BSpline



BSpline([0@0,1@1,3.2@2,4@4]).gui


(

b = BSpline([0@0,1@1,3.2@2,4@4]);

b.gui(nil,1000@200);

)


// returns points spaced evenly along the spline in 2D space

b.interpolate(512)


// to use X as time we need y values spaced in even X units

b.interpolateAlongX(512).plot2



(

// in some cases a spline's curvature can cause overlaps in time

// ie. the domain values are not monotonic

b = BSpline([ Point(0,0), Point(0.90534979423868, 2.675), Point(1.6615226337449, 0.725), Point(4, 4) ], 2.0);

b.gui;

// in this situation the overlap is ignored during interpolateAlongX

b.interpolateAlongX(512).plot2

)



// edited splines can be saved 

b.asCompileString



// playing it as a kr Envelope

(

s.boot;

b = BSpline([ Point(0,0), Point(0.58963874282376, 0.4134375), Point(2.2682499386103, 0.826875), Point(4.8180390967671, 0.013125) ], 2.3870967741935,false);

b.gui(nil,1000@200);


)


(

{

PinkNoise.ar * b.kr(doneAction:2)

}.play

)


// read through the spline with a position input

(

{

PinkNoise.ar * b.readKr(MouseX.kr(0,b.duration))

}.play

)



// looped spline

(

b = BSpline([ Point(0.65058398297395, 0.6098203125), Point(1.741086659197, 0.165375), Point(2.9369219802824, 0.9095625), Point(3.7238187977842, 0.2790703125) ], 6.3870967741935, true);

b.gui(nil,1000@200);


)

b.interpolateAlongX(128).plot2


(

{

PinkNoise.ar * b.ar(10)

}.play

)


(

{

b.ar(400)

}.play

)
